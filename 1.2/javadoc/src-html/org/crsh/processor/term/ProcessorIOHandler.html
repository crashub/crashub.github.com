020    package org.crsh.processor.term;
021    
022    import org.crsh.plugin.CRaSHPlugin;
023    import org.crsh.shell.Shell;
024    import org.crsh.shell.ShellFactory;
025    import org.crsh.term.console.ConsoleTerm;
026    import org.crsh.term.spi.TermIO;
027    import org.crsh.term.spi.TermIOHandler;
028    
029    import java.security.Principal;
030    
031    public class ProcessorIOHandler extends CRaSHPlugin<TermIOHandler> implements TermIOHandler {
032    
033      /** . */
034      private ShellFactory factory;
035    
036      @Override
037      public TermIOHandler getImplementation() {
038        return this;
039      }
040    
041      @Override
042      public void init() {
043        this.factory = getContext().getPlugin(ShellFactory.class);
044      }
045    
046      @Override
047      public void destroy() {
048      }
049    
050      public void handle(final TermIO io, Principal user) {
051        Shell shell = factory.create(user);
052        ConsoleTerm term = new ConsoleTerm(io);
053        Processor processor = new Processor(term, shell);
054        processor.addListener(io);
055        processor.addListener(shell);
056        processor.run();
057      }
058    }
