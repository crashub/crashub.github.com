020    package org.crsh.shell.impl.command;
021    
022    abstract class Token {
023    
024      public abstract String toString();
025    
026      public static Token EOF = new Token(){
027        @Override
028        public String toString() {
029          return "EOF";
030        }
031      };
032    
033      public static Token PIPE = new Token(){
034        @Override
035        public String toString() {
036          return "PIPE";
037        }
038      };
039    
040      public static class Command extends Token {
041    
042        /** . */
043        final String line;
044    
045        public Command(String line) {
046          this.line = line;
047        }
048    
049        @Override
050        public String toString() {
051          return "Command[" + line + "]";
052        }
053      }
054    }
