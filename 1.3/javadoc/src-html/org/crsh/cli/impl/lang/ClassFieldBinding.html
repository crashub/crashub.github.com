020    package org.crsh.cli.impl.lang;
021    
022    import org.crsh.cli.impl.invocation.InvocationException;
023    
024    import java.lang.reflect.Field;
025    
026    public class ClassFieldBinding implements Binding {
027    
028      /** . */
029      private final Field field;
030    
031      public ClassFieldBinding(Field field) {
032        this.field = field;
033      }
034    
035      public Field getField() {
036        return field;
037      }
038    
039      @Override
040      public void set(Object target, Object[] args, Object value) throws InvocationException {
041        try {
042          field.setAccessible(true);
043          field.set(target, value);
044        }
045        catch (Exception e) {
046          throw new InvocationException(e.getMessage(), e);
047        }
048      }
049    }
