019    package org.crsh.lang.impl.groovy.command;
020    
021    import groovy.lang.Binding;
022    import groovy.lang.MissingPropertyException;
023    import groovy.lang.Script;
024    
025    /** @author <a href="mailto:julien.viet@exoplatform.com">Julien Viet</a> */
026    public abstract class GroovyScript extends Script {
027    
028      protected GroovyScript() {
029      }
030    
031      protected GroovyScript(Binding binding) {
032        super(binding);
033      }
034    
035      @Override
036      public Object getProperty(String property) {
037        try {
038          return super.getProperty(property);
039        }
040        catch (MissingPropertyException e) {
041          return null;
042        }
043      }
044    }
