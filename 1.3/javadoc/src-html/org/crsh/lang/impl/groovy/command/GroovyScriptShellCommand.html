019package org.crsh.lang.impl.groovy.command;
020
021import groovy.lang.Binding;
022import groovy.lang.Closure;
023import org.crsh.cli.descriptor.CommandDescriptor;
024import org.crsh.cli.impl.descriptor.HelpDescriptor;
025import org.crsh.cli.impl.descriptor.IntrospectionException;
026import org.crsh.cli.impl.invocation.InvocationMatch;
027import org.crsh.cli.impl.lang.CommandFactory;
028import org.crsh.cli.impl.lang.Instance;
029import org.crsh.cli.spi.Completer;
030import org.crsh.command.CommandContext;
031import org.crsh.groovy.GroovyCommand;
032import org.crsh.shell.ErrorKind;
033import org.crsh.shell.impl.command.spi.CommandException;
034import org.crsh.lang.impl.groovy.ast.ScriptLastStatementTransformer;
035import org.crsh.shell.impl.command.spi.CommandMatch;
036import org.crsh.shell.impl.command.spi.CommandInvoker;
037import org.crsh.shell.impl.command.InvocationContextImpl;
038import org.crsh.command.RuntimeContext;
039import org.crsh.shell.impl.command.spi.Command;
040import org.crsh.util.Utils;
041
042import java.io.IOException;
043import java.lang.reflect.UndeclaredThrowableException;
044import java.util.List;
045
046/** @author Julien Viet */
047public class GroovyScriptShellCommand<T extends GroovyScriptCommand> extends Command<Instance<T>> {
048
049  /** . */
050  private final Class<T> clazz;
051
052  /** . */
053  private final boolean hasExplicitReturn;
054
055  /** . */
056  private final CommandDescriptor<Instance<T>> descriptor;
057
058  public GroovyScriptShellCommand(Class<T> clazz) throws IntrospectionException {
059
060    //
061    CommandFactory factory = new CommandFactory(getClass().getClassLoader());
062
063    boolean hasExplicitReturn;
064    try {
065      clazz.getDeclaredField(ScriptLastStatementTransformer.FIELD_NAME);
066      hasExplicitReturn = true;
067    }
068    catch (NoSuchFieldException e) {
069      hasExplicitReturn = false;
070    }
071
072    //
073    this.clazz = clazz;
074    this.descriptor = HelpDescriptor.create(factory.create(clazz));
075    this.hasExplicitReturn = hasExplicitReturn;
076  }
077
078  @Override
079  public CommandDescriptor<Instance<T>> getDescriptor() {
080    return descriptor;
081  }
082
083  @Override
084  protected CommandMatch<?, ?> resolve(final InvocationMatch<Instance<T>> match) {
085    return new CommandMatch<Void, Object>() {
086      @Override
087      public CommandInvoker<Void, Object> getInvoker() throws CommandException {
088        List<String> chunks = Utils.chunks(match.getRest());
089        String[] args = chunks.toArray(new String[chunks.size()]);
090        return GroovyScriptShellCommand.this.getInvoker(args);
091      }
092
093      @Override
094      public Class<Object> getProducedType() {
095        return Object.class;
096      }
097
098      @Override
099      public Class<Void> getConsumedType() {
100        return Void.class;
101      }
102    };
103  }
104
105  private T createCommand() throws CommandException {
106    T command;
107    try {
108      command = clazz.newInstance();
109    }
110    catch (Exception e) {
111      String name = clazz.getSimpleName();
112      throw new CommandException(name, ErrorKind.INTERNAL, "Could not create command " + name + " instance", e);
113    }
114    return command;
115  }
116
117  @Override
118  protected Completer getCompleter(RuntimeContext context) throws CommandException {
119    return null;
120  }
121
122  private CommandInvoker<Void, Object> getInvoker(final String[] args) throws CommandException {
123    final T command = createCommand();
124    return new CommandInvoker<Void, Object>() {
125
126      /** . */
127      private org.crsh.command.InvocationContext<Object> context;
128
129      public final Class<Object> getProducedType() {
130        return Object.class;
131      }
132
133      public final Class<Void> getConsumedType() {
134        return Void.class;
135      }
136
137      public void open(CommandContext<? super Object> consumer) {
138
139        // Set the context
140        context = new InvocationContextImpl<Object>((CommandContext<Object>)consumer);
141
142        // Set up current binding
143        Binding binding = new Binding(consumer.getSession());
144
145        // Set the args on the script
146        binding.setProperty("args", args);
147
148        //
149        command.setBinding(binding);
150
151
152        //
153        command.pushContext(context);
154
155        //
156        try {
157          //
158          Object ret = command.run();
159
160          // Evaluate the closure
161          if (ret instanceof Closure) {
162            Closure closure = (Closure)ret;
163            ret = closure.call(args);
164          }
165
166          //
167          if (ret != null) {
168            if (hasExplicitReturn) {
169              context.provide(ret);
170            }
171          }
172        }
173        catch (Exception t) {
174          throw GroovyCommand.unwrap(t);
175        }
176      }
177
178      public void provide(Void element) throws IOException {
179        // Should never be called
180      }
181
182      public void flush() throws IOException {
183        context.flush();
184      }
185
186      public void close() throws IOException, UndeclaredThrowableException {
187        context = null;
188        command.popContext();
189      }
190    };
191  }
