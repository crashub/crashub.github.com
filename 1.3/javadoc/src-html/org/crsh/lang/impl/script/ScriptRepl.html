037/** @author Julien Viet */
038public class ScriptRepl implements Repl {
039
040  /** . */
041  private static final ScriptRepl instance = new ScriptRepl();
042
043  /** . */
044  static final Logger log = Logger.getLogger(ScriptRepl.class.getName());
045
046  public static ScriptRepl getInstance() {
047    return instance;
048  }
049
050  /** . */
051  private final Language lang = new Language() {
052    @Override public String getName() { return "script"; }
053    @Override public String getDisplayName() { return "Script 1.0"; }
054    @Override public boolean isActive() { return true; }
055    @Override public Repl getRepl() { return ScriptRepl.this; }
056    @Override public org.crsh.lang.spi.Compiler getCompiler() { return ScriptCompiler.instance; }
057    @Override public void init(ShellSession session) { }
058    @Override public void destroy(ShellSession session) { }
059  };
060
061  private ScriptRepl() {
062  }
063
064  @Override
065  public Language getLanguage() {
066    return lang;
067  }
068
069  public String getName() {
070    return "script";
071  }
072
073  @Override
074  public String getDescription() {
075    return "The Script repl provides command line interpreter with a bash like syntax";
076  }
077
078  public ReplResponse eval(ShellSession session, String request) {
079    PipeLineFactory factory;
080    try {
081      factory = Token.parse(request).createFactory();
082    }
083    catch (CommandException e) {
084      return new ReplResponse.Response(ShellResponse.error(e.getErrorKind(), e.getMessage(), e.getCause()));
085    }
086    if (factory != null) {
087      try {
088        CommandInvoker<Void, Object> invoker = factory.create(session);
089        return new ReplResponse.Invoke(invoker);
090      }
091      catch (CommandException e) {
092        log.log(Level.FINER, "Could not create command", e);
093        return new ReplResponse.Response(ShellResponse.unknownCommand(e.getCommandName()));
094      }
095    } else {
096      return new ReplResponse.Response(ShellResponse.noCommand());
097    }
098  }
099
100  public CompletionMatch complete(ShellSession session, String prefix) {
101    Token ast = Token.parse(prefix);
102    String termPrefix;
103    if (ast != null) {
104      Token last = ast.getLast();
105      termPrefix = Utils.trimLeft(last.value);
106    } else {
107      termPrefix = "";
108    }
109
110    //
111    log.log(Level.FINE, "Retained term prefix is " + termPrefix);
112    CompletionMatch completion;
113    int pos = termPrefix.indexOf(' ');
114    if (pos == -1) {
115      Completion.Builder builder = Completion.builder(termPrefix);
116      for (Map.Entry<String, String> command : session.getCommands()) {
117        String name = command.getKey();
118        if (name.startsWith(termPrefix)) {
119          builder.add(name.substring(termPrefix.length()), true);
120        }
121      }
122      completion = new CompletionMatch(Delimiter.EMPTY, builder.build());
123    } else {
124      String commandName = termPrefix.substring(0, pos);
125      termPrefix = termPrefix.substring(pos);
126      try {
127        Command<?> command = session.getCommand(commandName);
128        if (command != null) {
129          completion = command.complete(new RuntimeContextImpl(session, session.getContext().getAttributes()), termPrefix);
130        } else {
131          completion = new CompletionMatch(Delimiter.EMPTY, Completion.create());
132        }
133      }
134      catch (CommandException e) {
135        log.log(Level.FINE, "Could not create command for completion of " + prefix, e);
136        completion = new CompletionMatch(Delimiter.EMPTY, Completion.create());
137      }
138    }
139
140    //
141    return completion;
142  }
143}
