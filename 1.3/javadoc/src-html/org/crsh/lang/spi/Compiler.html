019    package org.crsh.lang.spi;
020    
021    import org.crsh.shell.impl.command.ShellSession;
022    import org.crsh.shell.impl.command.spi.CommandException;
023    
024    import java.util.Set;
025    
026    /**
027     *
028     *
029     * @author Julien Viet
030     */
031    public interface Compiler {
032    
033      /**
034       * Returns the set of extensions managed by this implementation.
035       *
036       * @return the set of extensions, for instance ("groovy")
037       */
038      Set<String> getExtensions();
039    
040      /**
041       * Compile a command..
042       *
043       * @param name the command name
044       * @param source the command source  @return the command or null if no command can be resolved
045       * @throws org.crsh.shell.impl.command.spi.CommandException when the command exists but cannot be created
046       * @throws NullPointerException if the command name is null
047       */
048      CommandResolution compileCommand(String name, byte[] source) throws CommandException, NullPointerException;
049    
050      //
051      String doCallBack(ShellSession session, String name, String defaultValue);
052    
053    }
