019package org.crsh.plugin;
020
021import java.util.ArrayList;
022import java.util.ServiceConfigurationError;
023import java.util.ServiceLoader;
024import java.util.logging.Level;
025import java.util.logging.Logger;
026
027public class ServiceLoaderDiscovery implements PluginDiscovery {
028
029  /** . */
030  private static final Logger log = Logger.getLogger(PluginManager.class.getName());
031
032  /** . */
033  private final ClassLoader classLoader;
034
035  /**
036   * Create a new instance.
037   *
038   * @param classLoader the loader for the discovery
039   * @throws NullPointerException if the loader argument is null
040   */
041  public ServiceLoaderDiscovery(ClassLoader classLoader) throws NullPointerException {
042    if (classLoader == null) {
043      throw new NullPointerException();
044    }
045
046    //
047    this.classLoader = classLoader;
048  }
049
050  public Iterable<CRaSHPlugin<?>> getPlugins() {
051
052    //
053    ArrayList<CRaSHPlugin<?>> plugins = new ArrayList<CRaSHPlugin<?>>();
054    try {
055      ServiceLoader<CRaSHPlugin> loader = ServiceLoader.load(CRaSHPlugin.class, classLoader);
056      for (CRaSHPlugin<?> plugin : loader) {
057        log.log(Level.INFO, "Loaded plugin " + plugin);
058        plugins.add(plugin);
059      }
060    }
061    catch (ServiceConfigurationError e) {
062      log.log(Level.SEVERE, "Could not load plugins", e);
063    }
064
065    //
066    return plugins;
067  }
068}
