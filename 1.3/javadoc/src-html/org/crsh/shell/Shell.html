020    package org.crsh.shell;
021    
022    import org.crsh.cli.impl.completion.CompletionMatch;
023    
024    import java.io.Closeable;
025    
026    public interface Shell extends Closeable {
027    
028      /**
029       * Returns the welcome message.
030       *
031       * @return the welcome message
032       */
033      String getWelcome();
034    
035      /**
036       * Returns the shell prompt.
037       *
038       * @return the shell prompt
039       */
040      String getPrompt();
041    
042      /**
043       * Process a request.
044       *
045       * @param request the request to process
046       * @return the process
047       * @throws IllegalStateException if the shell cannot create a process
048       */
049      ShellProcess createProcess(String request) throws IllegalStateException;
050    
051      /**
052       * Completion.
053       *
054       *
055       * @param prefix the prefix to complete
056       * @return the sorted list of available suffixes
057       */
058      CompletionMatch complete(String prefix);
059    
060    }
