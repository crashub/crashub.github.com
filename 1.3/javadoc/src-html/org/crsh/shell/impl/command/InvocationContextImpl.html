020package org.crsh.shell.impl.command;
021
022import org.crsh.command.CommandContext;
023import org.crsh.text.Screenable;
024import org.crsh.shell.impl.command.spi.CommandException;
025import org.crsh.command.InvocationContext;
026import org.crsh.command.ScriptException;
027import org.crsh.lang.impl.script.Token;
028import org.crsh.text.ScreenContext;
029import org.crsh.lang.impl.script.PipeLineFactory;
030import org.crsh.shell.impl.command.spi.CommandInvoker;
031import org.crsh.text.RenderPrintWriter;
032import org.crsh.text.Style;
033import org.crsh.util.Utils;
034
035import java.io.IOException;
036import java.util.Map;
037
038public final class InvocationContextImpl<P> implements InvocationContext<P> {
039
040  /** . */
041  private static final int WRITTEN = 0;
042
043  /** . */
044  private static final int FLUSHED = 1;
045
046  /** . */
047  private static final int CLOSED = 2;
048
049  /** . */
050  private final CommandContext<P> commandContext;
051
052  /** . */
053  private RenderPrintWriter writer;
054
055  /** . */
056  int status;
057
058  public InvocationContextImpl(CommandContext<P> commandContext) {
059    this.commandContext = commandContext;
060    this.status = FLUSHED;
061  }
062
063  public RenderPrintWriter getWriter() {
064    if (writer == null) {
065      writer = new RenderPrintWriter(new ScreenContext() {
066        public int getWidth() {
067          return InvocationContextImpl.this.getWidth();
068        }
069        public int getHeight() {
070          return InvocationContextImpl.this.getHeight();
071        }
072        public Screenable append(CharSequence s) throws IOException {
073          InvocationContextImpl.this.append(s);
074          return this;
075        }
076        public Appendable append(char c) throws IOException {
077          InvocationContextImpl.this.append(c);
078          return this;
079        }
080        public Appendable append(CharSequence csq, int start, int end) throws IOException {
081          InvocationContextImpl.this.append(csq, start, end);
082          return this;
083        }
084        public Screenable append(Style style) throws IOException {
085          InvocationContextImpl.this.append(style);
086          return this;
087        }
088        public Screenable cls() throws IOException {
089          InvocationContextImpl.this.cls();
090          return this;
091        }
092        public void flush() throws IOException {
093          InvocationContextImpl.this.flush();
094        }
095      });
096    }
097    return writer;
098  }
099
100  public boolean takeAlternateBuffer() throws IOException {
101    return commandContext.takeAlternateBuffer();
102  }
103
104  public boolean releaseAlternateBuffer() throws IOException {
105    return commandContext.releaseAlternateBuffer();
106  }
107
108  public CommandInvoker<?, ?> resolve(String s) throws ScriptException, IOException {
109    CRaSHSession session = (CRaSHSession)getSession();
110    Token token2 = Token.parse(s);
111    try {
112      PipeLineFactory factory = token2.createFactory();
113      return factory.create(session);
114    }
115    catch (CommandException e) {
116      throw new ScriptException(e);
117    }
118  }
119
120  public Class<P> getConsumedType() {
121    return commandContext.getConsumedType();
122  }
123
124  public String getProperty(String propertyName) {
125    return commandContext.getProperty(propertyName);
126  }
127
128  public String readLine(String msg, boolean echo) throws IOException, InterruptedException {
129    return commandContext.readLine(msg, echo);
130  }
131
132  public int getWidth() {
133    return commandContext.getWidth();
134  }
135
136  public int getHeight() {
137    return commandContext.getHeight();
138  }
139
140  public Screenable append(CharSequence s) throws IOException {
141    if (status != CLOSED) {
142      status = WRITTEN;
143      commandContext.append(s);
144    }
145    return this;
146  }
147
148  public Screenable append(char c) throws IOException {
149    if (status != CLOSED) {
150      status = WRITTEN;
151      commandContext.append(c);
152    }
153    return this;
154  }
155
156  public Screenable append(CharSequence csq, int start, int end) throws IOException {
157    if (status != CLOSED) {
158      status = WRITTEN;
159      commandContext.append(csq, start, end);
160    }
161    return this;
162  }
163
164  public Screenable append(Style style) throws IOException {
165    if (status != CLOSED) {
166      status = WRITTEN;
167      commandContext.append(style);
168    }
169    return this;
170  }
171
172  public Screenable cls() throws IOException {
173    if (status != CLOSED) {
174      status = WRITTEN;
175      commandContext.cls();
176    }
177    return this;
178  }
179
180  public void provide(P element) throws IOException {
181    if (status != CLOSED) {
182      status = WRITTEN;
183      commandContext.provide(element);
184    }
185  }
186
187  public void flush() throws IOException {
188    if (status == WRITTEN) {
189      status = FLUSHED;
190      commandContext.flush();
191    }
192  }
193
194  public void close() throws IOException {
195    if (status != CLOSED) {
196      Utils.flush(this);
197      status = CLOSED;
198      Utils.close(commandContext);
199    }
200  }
201
202  public Map<String, Object> getSession() {
203    return commandContext.getSession();
204  }
205
206  public Map<String, Object> getAttributes() {
207    return commandContext.getAttributes();
208  }
209
210  public InvocationContextImpl<P> leftShift(Object o) throws IOException {
211    Class<P> consumedType = getConsumedType();
212    if (consumedType.isInstance(o)) {
213      P p = consumedType.cast(o);
214      provide(p);
215    }
216    return this;
217  }
218}
