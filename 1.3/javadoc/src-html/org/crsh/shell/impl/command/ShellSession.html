019package org.crsh.shell.impl.command;
020
021import org.crsh.lang.spi.Repl;
022import org.crsh.shell.impl.command.spi.Command;
023import org.crsh.shell.impl.command.spi.CommandException;
024import org.crsh.plugin.PluginContext;
025
026import java.util.Map;
027
028/** @author Julien Viet */
029public interface ShellSession extends Map<String, Object> {
030
031  PluginContext getContext();
032
033  Iterable<Map.Entry<String, String>> getCommands();
034
035  Command<?> getCommand(String name) throws CommandException;
036
037  /**
038   * @return the current repl of this session
039   */
040  Repl getRepl();
041
042  /**
043   * Set the current repl of this session.
044   *
045   * @param repl the new repl
046   * @throws NullPointerException if the repl is null
047   */
048  void setRepl(Repl repl) throws NullPointerException;
049
050}
