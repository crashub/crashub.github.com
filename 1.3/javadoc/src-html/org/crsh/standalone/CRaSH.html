085  private void copyCmd(org.crsh.vfs.File src, File dst) throws IOException {
086    if (src.hasChildren()) {
087      if (!dst.exists()) {
088        if (dst.mkdir()) {
089          log.fine("Could not create dir " + dst.getCanonicalPath());
090        }
091      }
092      if (dst.exists() && dst.isDirectory()) {
093        for (org.crsh.vfs.File child : src.children()) {
094          copyCmd(child, new File(dst, child.getName()));
095        }
096      }
097    } else {
098      if (!dst.exists()) {
099        Resource resource = src.getResource();
100        if (resource != null) {
101          log.info("Copied command " + src.getPath().getValue() + " to " + dst.getCanonicalPath());
102          Utils.copy(new ByteArrayInputStream(resource.getContent()), new FileOutputStream(dst));
103        }
104      }
105    }
106  }
107
108  private void copyConf(org.crsh.vfs.File src, File dst) throws IOException {
109    if (!src.hasChildren()) {
110      if (!dst.exists()) {
111        Resource resource = ResourceManager.loadConf(src);
112        if (resource != null) {
113          log.info("Copied resource " + src.getPath().getValue() + " to " + dst.getCanonicalPath());
114          Utils.copy(new ByteArrayInputStream(resource.getContent()), new FileOutputStream(dst));
115        }
116      }
117    }
118  }

120  private String toString(FS.Builder builder) {
121    StringBuilder sb = new StringBuilder();
122    List<Mount<?>> mounts = builder.getMounts();
123    for (int i = 0;i < mounts.size();i++) {
124      Mount<?> mount = mounts.get(i);
125      if (i > 0) {
126        sb.append(';');
127      }
128      sb.append(mount.getValue());
129    }
130    return sb.toString();
131  }
132
133  private FS.Builder createBuilder() throws IOException {
134    FileMountFactory fileDriver = new FileMountFactory(Utils.getCurrentDirectory());
135    ClassPathMountFactory classpathDriver = new ClassPathMountFactory(Thread.currentThread().getContextClassLoader());
136    return new FS.Builder().register("file", fileDriver).register("classpath", classpathDriver);
137  }

139  @Command
140  public void main(
141    @Option(names= {"non-interactive"})
142    @Usage("non interactive mode, the JVM io will not be used")
143    Boolean nonInteractive,
144    @Option(names={"c","cmd"})
145    @Usage("the command mounts")
146    String cmd,
147    @Option(names={"conf"})
148    @Usage("the conf mounts")
149    String conf,
150    @Option(names={"p","property"})
151    @Usage("set a property of the form a=b")
152    List<String> properties,
153    @Option(names = {"cmd-folder"})
154    @Usage("a folder in which commands should be extracted")
155    String cmdFolder,
156    @Option(names = {"conf-folder"})
157    @Usage("a folder in which configuration should be extracted")
158    String confFolder,
159    @Argument(name = "pid")
160    @Usage("the optional list of JVM process id to attach to")
161    List<Integer> pids) throws Exception {
162
163    //
164    boolean interactive = nonInteractive == null || !nonInteractive;
165
166    //
167    if (conf == null) {
168      conf = "classpath:/crash/";
169    }
170    FS.Builder confBuilder = createBuilder().mount(conf);
171    if (confFolder != null) {
172      File dst = new File(confFolder);
173      if (!dst.isDirectory()) {
174        throw new Exception("Directory " + dst.getAbsolutePath() + " does not exist");
175      }
176      org.crsh.vfs.File f = confBuilder.build().get(Path.get("/"));
177      log.info("Extracting conf resources to " + dst.getAbsolutePath());
178      for (org.crsh.vfs.File child : f.children()) {
179        if (!child.hasChildren()) {
180          copyConf(child, new File(dst, child.getName()));
181        }
182      }
183      confBuilder = createBuilder().mount("file:" + dst.getAbsolutePath());
184    }
185
186    //
187    if (cmd == null) {
188      cmd = "classpath:/crash/commands/";
189    }
190    FS.Builder cmdBuilder = createBuilder().mount(cmd);
191    if (cmdFolder != null) {
192      File dst = new File(cmdFolder);
193      if (!dst.isDirectory()) {
194        throw new Exception("Directory " + dst.getAbsolutePath() + " does not exist");
195      }
196      org.crsh.vfs.File f = cmdBuilder.build().get(Path.get("/"));
197      log.info("Extracting command resources to " + dst.getAbsolutePath());
198      copyCmd(f, dst);
199      cmdBuilder = createBuilder().mount("file:" + dst.getAbsolutePath());
200    }
201
202    //
203    CloseableList closeable = new CloseableList();
204    Shell shell;
205    if (pids != null && pids.size() > 0) {
206
207      //
208      if (interactive && pids.size() > 1) {
209        throw new Exception("Cannot attach to more than one JVM in interactive mode");
210      }
211
212      // Compute classpath
213      String classpath = System.getProperty("java.class.path");
214      String sep = System.getProperty("path.separator");
215      StringBuilder buffer = new StringBuilder();
216      for (String path : classpath.split(Pattern.quote(sep))) {
217        File file = new File(path);
218        if (file.exists()) {
219          if (buffer.length() > 0) {
220            buffer.append(' ');
221          }
222          buffer.append(file.getCanonicalPath());
223        }
224      }
225
226      // Create manifest
227      Manifest manifest = new Manifest();
228      Attributes attributes = manifest.getMainAttributes();
229      attributes.putValue("Agent-Class", Agent.class.getName());
230      attributes.put(Attributes.Name.MANIFEST_VERSION, "1.0");
231      attributes.put(Attributes.Name.CLASS_PATH, buffer.toString());
232
233      // Create jar file
234      File agentFile = File.createTempFile("agent", ".jar");
235      agentFile.deleteOnExit();
236      JarOutputStream out = new JarOutputStream(new FileOutputStream(agentFile), manifest);
237      out.close();
238      log.log(Level.INFO, "Created agent jar " + agentFile.getCanonicalPath());
239
240      // Build the options
241      StringBuilder sb = new StringBuilder();
242
243      // Path configuration
244      sb.append("--cmd ");
245      Delimiter.EMPTY.escape(toString(cmdBuilder), sb);
246      sb.append(' ');
247      sb.append("--conf ");
248      Delimiter.EMPTY.escape(toString(confBuilder), sb);
249      sb.append(' ');
250
251      // Propagate canonical config
252      if (properties != null) {
253        for (String property : properties) {
254          sb.append("--property ");
255          Delimiter.EMPTY.escape(property, sb);
256          sb.append(' ');
257        }
258      }
259
260      //
261      if (interactive) {
262        RemoteServer server = new RemoteServer(0);
263        int port = server.bind();
264        log.log(Level.INFO, "Callback server set on port " + port);
265        sb.append(port);
266        String options = sb.toString();
267        Integer pid = pids.get(0);
268        final VirtualMachine vm = VirtualMachine.attach("" + pid);
269        log.log(Level.INFO, "Loading agent with command " + options + " as agent " + agentFile.getCanonicalPath());
270        vm.loadAgent(agentFile.getCanonicalPath(), options);
271        server.accept();
272        shell = server.getShell();
273        closeable.add(new Closeable() {
274          public void close() throws IOException {
275            vm.detach();
276          }
277        });
278      } else {
279        for (Integer pid : pids) {
280          log.log(Level.INFO, "Attaching to remote process " + pid);
281          VirtualMachine vm = VirtualMachine.attach("" + pid);
282          String options = sb.toString();
283          log.log(Level.INFO, "Loading agent with command " + options + " as agent " + agentFile.getCanonicalPath());
284          vm.loadAgent(agentFile.getCanonicalPath(), options);
285        }
286        shell = null;
287      }
288    } else {
289      final Bootstrap bootstrap = new Bootstrap(
290          Thread.currentThread().getContextClassLoader(),
291          confBuilder.build(),
292          cmdBuilder.build());
293
294      //
295      if (properties != null) {
296        Properties config = new Properties();
297        for (String property : properties) {
298          int index = property.indexOf('=');
299          if (index == -1) {
300            config.setProperty(property, "");
301          } else {
302            config.setProperty(property.substring(0, index), property.substring(index + 1));
303          }
304        }
305        bootstrap.setConfig(config);
306      }
307
308      // Register shutdown hook
309      Runtime.getRuntime().addShutdownHook(new Thread() {
310        @Override
311        public void run() {
312          // Should trigger some kind of run interruption
313        }
314      });
315
316      // Do bootstrap
317      bootstrap.bootstrap();
318      Runtime.getRuntime().addShutdownHook(new Thread(){
319        @Override
320        public void run() {
321          bootstrap.shutdown();
322        }
323      });
324
325      //
326      if (interactive) {
327        ShellFactory factory = bootstrap.getContext().getPlugin(ShellFactory.class);
328        shell = factory.create(null);
329      } else {
330        shell = null;
331      }
332      closeable = null;
333    }
334
335    //
336    if (shell != null) {
337
338
339      final Terminal term = TerminalFactory.create();
340      final boolean installWindows = term.isAnsiSupported() && term instanceof AnsiWindowsTerminal;
341
342      //
343      if (installWindows) {
344        log.info("Installing windowd ansi console");
345        AnsiConsole.systemInstall();
346      }
347
348      //
349      Runtime.getRuntime().addShutdownHook(new Thread() {
350        @Override
351        public void run() {
352          if (installWindows) {
353            AnsiConsole.systemUninstall();
354          }
355          try {
356            term.restore();
357          }
358          catch (Exception ignore) {
359          }
360        }
361      });
362
363      // Use AnsiConsole only if term doesn't support Ansi
364      PrintStream out;
365      PrintStream err;
366      if (!term.isAnsiSupported()) {
367        out = AnsiConsole.out;
368        err = AnsiConsole.err;
369      } else {
370        out = new PrintStream(new BufferedOutputStream(new FileOutputStream(FileDescriptor.out), 16384), false);
371        err = new PrintStream(new BufferedOutputStream(new FileOutputStream(FileDescriptor.err), 16384), false);
372      }
373
374      //
375      FileInputStream in = new FileInputStream(FileDescriptor.in);
376      ConsoleReader reader = new ConsoleReader(null, in, out, term);
377
378      //
379      final JLineProcessor processor = new JLineProcessor(shell, reader, out);
380
381      //
382      InterruptHandler interruptHandler = new InterruptHandler(new Runnable() {
383        @Override
384        public void run() {
385          processor.interrupt();
386        }
387      });
388      interruptHandler.install();
389
390      //
391      Thread thread = new Thread(processor);
392      thread.setDaemon(true);
393      thread.start();
394
395      //
396      try {
397        processor.closed();
398      }
399      catch (Throwable t) {
400        t.printStackTrace();
401      }
402      finally {
403
404        //
405        if (closeable != null) {
406          Utils.close(closeable);
407        }
408
409        // Force exit
410        System.exit(0);
411      }
412    }
413  }
414
415  public static void main(String[] args) throws Exception {
416
417    StringBuilder line = new StringBuilder();
418    for (int i = 0;i < args.length;i++) {
419      if (i  > 0) {
420        line.append(' ');
421      }
422      Delimiter.EMPTY.escape(args[i], line);
423    }
424
425    //
426    CRaSH main = new CRaSH();
427    InvocationMatcher<Instance<CRaSH>> matcher = main.descriptor.matcher();
428    InvocationMatch<Instance<CRaSH>> match = matcher.parse(line.toString());
429    match.invoke(Util.wrap(main));
430  }
