020package org.crsh.stream;
021
022import java.io.IOException;
023
024/**
025 * Defines the interface for a consumer.
026 *
027 * @param <C> the consumed element generic type
028 */
029public interface Consumer<C> {
030
031  /**
032   * Provide an element.
033   *
034   * @param element the provided element
035   * @throws Exception any exception
036   */
037  void provide(C element) throws Exception;
038
039  /**
040   * Flush the stream.
041   *
042   * @throws IOException any io exception
043   */
044  void flush() throws IOException;
045
046  /**
047   * Returns the class of the element generic type.
048   *
049   * @return the consumed type
050   */
051  Class<C> getConsumedType();
052}
