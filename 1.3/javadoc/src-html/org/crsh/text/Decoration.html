020package org.crsh.text;
021
022public enum Decoration {
023
024  /** . */
025  bold(0, true, "1"),
026
027  /** . */
028  bold_off(0, false, "22"),
029
030  /** . */
031  underline(1, true, "4"),
032
033  /** . */
034  underline_off(1, false, "24"),
035
036  /** . */
037  blink(2, true, "5"),
038
039  /** . */
040  blink_off(2, false, "25");
041
042  /** . */
043  final int index;
044
045  /** . */
046  final boolean on;
047
048  /** . */
049  public final String code;
050
051  private Decoration(int index, boolean on, String code) {
052    this.index = index;
053    this.on = on;
054    this.code = code;
055  }
056
057  public Style.Composite fg(Color value) {
058    return foreground(value);
059  }
060
061  public Style.Composite foreground(Color value) {
062    return Style.style(this).foreground(value);
063  }
064
065  public Style.Composite bg(Color value) {
066    return background(value);
067  }
068
069  public Style.Composite background(Color value) {
070    return Style.style(this).background(value);
071  }
072
073  public Style.Composite bold() {
074    return bold(true);
075  }
076
077  public Style.Composite underline() {
078    return underline(true);
079  }
080
081  public Style.Composite blink() {
082    return blink(true);
083  }
084
085  public Style.Composite bold(Boolean value) {
086    return Style.style(this).bold(value);
087  }
088
089  public Style.Composite underline(Boolean value) {
090    return Style.style(this).underline(value);
091  }
092
093  public Style.Composite blink(Boolean value) {
094    return Style.style(this).blink(value);
095  }
096}
