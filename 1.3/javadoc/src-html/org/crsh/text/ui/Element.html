020package org.crsh.text.ui;
021
022import org.crsh.text.LineReader;
023import org.crsh.text.LineRenderer;
024import org.crsh.text.RenderAppendable;
025import org.crsh.text.Style;
026
027public abstract class Element {
028
029  /** . */
030  private Style.Composite style;
031
032  protected Element() {
033    this.style = null;
034  }
035
036  public abstract LineRenderer renderer();
037
038  public void render(RenderAppendable to) {
039    LineRenderer renderer = renderer();
040
041    // For now height - 1 because of the char that goes to the line in some impl
042    LineReader reader = renderer.reader(to.getWidth(), to.getHeight() - 1);
043    if (reader != null) {
044      while (reader.hasLine()) {
045        reader.renderLine(to);
046        to.append('\n');
047      }
048    }
049  }
050
051  public final Style.Composite getStyle() {
052    return style;
053  }
054
055  public final void setStyle(Style.Composite style) {
056    this.style = style;
057  }
058
059  public Element style(Style.Composite style) {
060    setStyle(style);
061    return this;
062  }
063
064  public static RowElement row() {
065    return new RowElement();
066  }
067
068  public static RowElement header() {
069    return new RowElement(true);
070  }
071
072  public static LabelElement label(String value) {
073    return new LabelElement(value);
074  }
075}
