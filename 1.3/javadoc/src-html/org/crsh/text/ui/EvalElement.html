020package org.crsh.text.ui;
021
022import groovy.lang.Closure;
023import org.crsh.groovy.GroovyCommand;
024import org.crsh.text.Screenable;
025import org.crsh.shell.impl.command.spi.CommandInvoker;
026import org.crsh.lang.impl.groovy.command.GroovyScriptCommand;
027import org.crsh.command.InvocationContext;
028import org.crsh.command.ScriptException;
029import org.crsh.text.CLS;
030import org.crsh.text.LineRenderer;
031import org.crsh.text.RenderPrintWriter;
032import org.crsh.text.Renderer;
033import org.crsh.text.Style;
034
035import java.io.IOException;
036import java.util.LinkedList;
037import java.util.Map;
038
039public class EvalElement extends Element {
040
041  /** The closure to evaluate. */
042  Closure closure;
043
044  public LineRenderer renderer() {
045
046    Object owner = closure.getOwner();
047
048    //
049    final InvocationContext ctx;
050    Object cmd;
051    while (true) {
052      if (owner instanceof GroovyCommand) {
053        cmd = owner;
054        ctx = ((GroovyCommand)cmd).peekContext();
055        break;
056      } else if (owner instanceof GroovyScriptCommand) {
057        cmd = owner;
058        ctx = ((GroovyScriptCommand)cmd).peekContext();
059        break;
060      } else if (owner instanceof Closure) {
061        owner = ((Closure)owner).getOwner();
062      } else {
063        throw new UnsupportedOperationException("Cannot resolver owner " + owner + " to command");
064      }
065    }
066
067    //
068    final LinkedList<LineRenderer> renderers = new LinkedList<LineRenderer>();
069
070    //
071    final InvocationContext nested = new InvocationContext() {
072
073      /** . */
074      private LinkedList<Object> buffer = new LinkedList<Object>();
075
076      /** . */
077      private Renderer renderable;
078
079      public CommandInvoker<?, ?> resolve(String s) throws ScriptException, IOException {
080        return ctx.resolve(s);
081      }
082
083      public boolean takeAlternateBuffer() {
084        return false;
085      }
086
087      public boolean releaseAlternateBuffer() {
088        return false;
089      }
090
091      public RenderPrintWriter getWriter() {
092        return ctx.getWriter();
093      }
094
095      public Map<String, Object> getSession() {
096        return ctx.getSession();
097      }
098
099      public Map<String, Object> getAttributes() {
100        return ctx.getAttributes();
101      }
102
103      public int getWidth() {
104        return ctx.getWidth();
105      }
106
107      public int getHeight() {
108        return ctx.getHeight();
109      }
110
111      public String getProperty(String propertyName) {
112        return ctx.getProperty(propertyName);
113      }
114
115      public String readLine(String msg, boolean echo) {
116        return null;
117      }
118
119      public Class getConsumedType() {
120        return Object.class;
121      }
122
123      public Screenable append(CharSequence s) throws IOException {
124        provide(s);
125        return this;
126      }
127
128      @Override
129      public Appendable append(char c) throws IOException {
130        return append(Character.toString(c));
131      }
132
133      @Override
134      public Appendable append(CharSequence csq, int start, int end) throws IOException {
135        return append(csq.subSequence(start, end));
136      }
137
138      public Screenable append(Style style) throws IOException {
139        provide(style);
140        return this;
141      }
142
143      public Screenable cls() throws IOException {
144        provide(CLS.INSTANCE);
145        return this;
146      }
147
148      public void provide(Object element) throws IOException {
149        Renderer current = Renderer.getRenderable(element.getClass());
150        if (current == null) {
151          current = Renderer.ANY;
152        }
153        if (current != null) {
154          if (renderable != null && !current.equals(renderable)) {
155            flush();
156          }
157          buffer.addLast(element);
158          renderable = current;
159        }
160      }
161
162      public void flush() throws IOException {
163        // We don't really flush, we just compute renderables from the buffer
164        if (buffer.size() > 0) {
165          LineRenderer i = renderable.renderer(buffer.iterator());
166          buffer.clear();
167          renderers.add(i);
168        }
169      }
170
171      public void close() throws IOException {
172        // Nothing to do, except maybe release resources (and also prevent to do any other operation)
173      }
174    };
175
176    if (cmd instanceof GroovyCommand) {
177      ((GroovyCommand)cmd).pushContext(nested);
178    } else {
179      ((GroovyScriptCommand)cmd).pushContext(nested);
180    }
181    try {
182      closure.call();
183    }
184    finally {
185      if (cmd instanceof GroovyCommand) {
186        ((GroovyCommand)cmd).popContext();
187      } else {
188        ((GroovyScriptCommand)cmd).popContext();
189      }
190    }
191
192    // Be sure to flush
193    try {
194      nested.flush();
195    }
196    catch (IOException e) {
197      e.printStackTrace();
198    }
199
200    //
201    return LineRenderer.vertical(renderers);
202  }
203}
