020package org.crsh.vfs;
021
022import org.crsh.vfs.spi.FSDriver;
023import org.crsh.vfs.spi.FSMountFactory;
024import org.crsh.vfs.spi.Mount;
025import org.crsh.vfs.spi.file.FileDriver;
026import org.crsh.vfs.spi.url.ClassPathMountFactory;
027import org.crsh.vfs.spi.url.URLDriver;
028
029import java.io.IOException;
030import java.net.URISyntaxException;
031import java.net.URL;
032import java.util.ArrayList;
033import java.util.HashMap;
034import java.util.List;
035
036/**
037 * The file system provides a federated view of {@link org.crsh.vfs.spi.FSDriver} mounts.
038 */
039public class FS {
040
041  public static class Builder {
042
043    /** . */
044    private HashMap<String, FSMountFactory<?>> resolvers;
045
046    /** . */
047    private ArrayList<Mount<?>> mounts = new ArrayList<Mount<?>>();
048
049    public Builder() {
050      this.resolvers = new HashMap<String, FSMountFactory<?>>();
051    }
052
053    /**
054     * Register a resolver.
055     *
056     * @param name the registration name
057     * @param resolver the resolver implementation
058     */
059    public Builder register(String name, FSMountFactory<?> resolver) {
060      resolvers.put(name, resolver);
061      return this;
062    }
063
064    public Builder mount(String name, Path path) throws IOException, IllegalArgumentException {
065      FSMountFactory<?> resolver = resolvers.get(name);
066      if (resolver == null) {
067        throw new IllegalArgumentException("Unknown driver " + name);
068      } else {
069        Mount<?> mount = resolver.create(path);
070        mounts.add(mount);
071        return this;
072      }
073    }
074
075    public Builder mount(String mountPointConfig) throws IOException {
076      int prev = 0;
077      while (true) {
078        int next = mountPointConfig.indexOf(';', prev);
079        if (next == -1) {
080          next = mountPointConfig.length();
081        }
082        if (next > prev) {
083          String mount = mountPointConfig.substring(prev, next);
084          int index = mount.indexOf(':');
085          String name;
086          String path;
087          if (index == -1) {
088            name = "classpath";
089            path = mount;
090          } else {
091            name = mount.substring(0, index);
092            path = mount.substring(index + 1);
093          }
094          mount(name, Path.get(path));
095          prev = next + 1;
096        } else {
097          break;
098        }
099      }
100      return this;
101    }
102
103    public List<Mount<?>> getMounts() {
104      return mounts;
105    }
106
107    public FS build() throws IOException {
108      FS fs = new FS();
109      for (Mount<?> mount : mounts) {
110        fs.mount(mount.getDriver());
111      }
112      return fs;
113    }
114  }
115
116  /** . */
117  final List<FSDriver<?>> drivers;
118
119  public FS() {
120    this.drivers = new ArrayList<FSDriver<?>>();
121  }
122
123  public File get(Path path) throws IOException {
124    return new File(this, path);
125  }
126
127  public FS mount(FSDriver<?> driver) throws IOException {
128    if (driver == null) {
129      throw new NullPointerException();
130    }
131    drivers.add(driver);
132    return this;
133  }
134
135  public FS mount(java.io.File root) throws IOException {
136    return mount(new FileDriver(root));
137  }
138
139  public FS mount(ClassLoader cl, Path path) throws IOException, URISyntaxException {
140    if (cl == null) {
141      throw new NullPointerException();
142    } else {
143      return mount(new ClassPathMountFactory(cl).create(path).getDriver());
144    }
145  }
146
147  public FS mount(Class<?> clazz) throws IOException, URISyntaxException {
148    if (clazz == null) {
149      throw new NullPointerException();
150    }
151    URL url = clazz.getProtectionDomain().getCodeSource().getLocation();
152    URLDriver driver = new URLDriver();
153    driver.merge(url);
154    return mount(driver);
155  }
156}
