020    package org.crsh.jcr.command;
021    
022    import org.crsh.cli.type.ValueType;
023    
024    public class PathValueType extends ValueType<Path> {
025    
026      public PathValueType() throws NullPointerException {
027        super(Path.class, PathCompleter.class);
028      }
029    
030      @Override
031      public <S extends Path> S parse(Class<S> type, String s) throws Exception {
032        return type.cast(new Path(s));
033      }
034    }
