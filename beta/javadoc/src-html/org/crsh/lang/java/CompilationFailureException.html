019    package org.crsh.lang.java;
020    
021    import javax.tools.Diagnostic;
022    import javax.tools.JavaFileObject;
023    import java.util.List;
024    
025    /** @author Julien Viet */
026    public class CompilationFailureException extends Exception {
027    
028      /** . */
029      private final List<Diagnostic<? extends JavaFileObject>> errors;
030    
031      public CompilationFailureException(List<Diagnostic<? extends JavaFileObject>> errors) {
032        this.errors = errors;
033      }
034    
035      public List<Diagnostic<? extends JavaFileObject>> getErrors() {
036        return errors;
037      }
038    
039      @Override
040      public String getMessage() {
041        StringBuilder message = new StringBuilder();
042        for (Diagnostic<? extends JavaFileObject> error : errors) {
043          message.append(error.getMessage(null));
044          message.append("\n");
045        }
046        return message.toString();
047      }
048    }
