020    package org.crsh.shell;
021    
022    import org.crsh.text.Chunk;
023    
024    import java.io.Flushable;
025    import java.io.IOException;
026    
027    /**
028     * The screen context extends the consumer and add information about the screen.
029     */
030    public interface ScreenContext extends Flushable {
031    
032      /**
033       * Returns the screen width in chars. When the value is not positive it means
034       * the value could not be determined.
035       *
036       * @return the term width
037       */
038      int getWidth();
039    
040      /**
041       * Returns the screen height in chars. When the value is not positive it means
042       * the value could not be determined.
043       *
044       * @return the term height
045       */
046      int getHeight();
047    
048      /**
049       * Write a chunk to the screen.
050       *
051       * @param chunk the chunk
052       * @throws IOException any io exception
053       */
054      void write(Chunk chunk) throws IOException;
055    
056    }
