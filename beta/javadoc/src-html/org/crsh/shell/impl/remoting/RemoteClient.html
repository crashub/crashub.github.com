020    package org.crsh.shell.impl.remoting;
021    
022    import org.crsh.shell.Shell;
023    import org.crsh.util.AbstractSocketClient;
024    
025    import java.io.IOException;
026    import java.io.InputStream;
027    import java.io.OutputStream;
028    
029    public class RemoteClient extends AbstractSocketClient {
030    
031      /** . */
032      private ClientAutomaton automaton;
033    
034      /** . */
035      private final Shell shell;
036    
037      public RemoteClient(int port, Shell shell) {
038        super(port);
039    
040        //
041        this.shell = shell;
042      }
043    
044      @Override
045      protected void handle(InputStream in, OutputStream out) throws IOException {
046        this.automaton = new ClientAutomaton(in, out, shell).addCloseListener(this);;
047      }
048    
049      public Runnable getRunnable() {
050        return automaton;
051      }
052    }
