020    package org.crsh.command;
021    
022    import org.crsh.io.Consumer;
023    
024    import java.io.IOException;
025    
026    class PipeCommandProxy<C, P> implements CommandInvoker<C, P> {
027    
028      /** . */
029      private final CommandContext<P> innerContext;
030    
031      /** . */
032      private final CommandInvoker<C, P> delegate;
033    
034      /** . */
035      private final Consumer<C> next;
036    
037      PipeCommandProxy(CommandContext<P> innerContext, CommandInvoker<C, P> delegate, Consumer<C> next) {
038        this.innerContext = innerContext;
039        this.delegate = delegate;
040        this.next = next;
041      }
042    
043      void fire() {
044        open(innerContext);
045      }
046    
047      public Class<P> getProducedType() {
048        return delegate.getProducedType();
049      }
050    
051      public Class<C> getConsumedType() {
052        return delegate.getConsumedType();
053      }
054    
055      public void open(CommandContext<P> consumer) {
056        if (next != null && next instanceof PipeCommandProxy) {
057          ((PipeCommandProxy)next).fire();
058        }
059        delegate.open(consumer);
060      }
061    
062      public void provide(C element) throws ScriptException, IOException {
063        delegate.provide(element);
064      }
065    
066      public void flush() throws ScriptException, IOException {
067        delegate.flush();
068        if (next != null && next instanceof PipeCommand) {
069          ((PipeCommand)next).flush();
070        }
071      }
072    
073      public void close() throws ScriptException, IOException {
074        delegate.close();
075        if (next != null && next instanceof PipeCommand) {
076          ((PipeCommand)next).close();
077        }
078      }
079    }
