020    package org.crsh.io;
021    
022    import java.io.Closeable;
023    import java.io.IOException;
024    
025    /**
026     * A producer that produces elements in a specific consumer.
027     *
028     * @param <P> the produced element generic type
029     * @param <C> the consumer element generic type
030     */
031    public interface Producer<P, C extends Consumer<P>> extends Closeable {
032    
033      /**
034       * Returns the class of the produced type.
035       *
036       * @return the produced type
037       */
038      Class<P> getProducedType();
039    
040      /**
041       * Open the producer with the specified consumer.
042       *
043       * @param consumer the consumer
044       */
045      void open(C consumer);
046    
047      /**
048       * Close the producer.
049       */
050      void close() throws IOException;
051    
052    }
